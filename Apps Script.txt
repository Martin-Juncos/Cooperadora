const scriptProp = PropertiesService.getScriptProperties()

function initialSetup () {
  const activeSpreadsheet = SpreadsheetApp.getActiveSpreadsheet()
  scriptProp.setProperty('key', activeSpreadsheet.getId())
}

function normalizeEmail (email) {
  return String(email || '').trim().toLowerCase()
}

function isAllowedEmail (email) {
  return allowedEmails.indexOf(normalizeEmail(email)) !== -1
}

function jsonResponse (payload) {
  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON)
}

function verifyGoogleToken (idToken) {
  if (!idToken) return null

  try {
    const response = UrlFetchApp.fetch(
      'https://oauth2.googleapis.com/tokeninfo?id_token=' + encodeURIComponent(idToken),
      { muteHttpExceptions: true }
    )

    if (response.getResponseCode() !== 200) return null

    const payload = JSON.parse(response.getContentText())
    const email = normalizeEmail(payload.email)
    const emailVerified = String(payload.email_verified) === 'true' || payload.email_verified === true

    if (payload.aud !== googleClientId) return null
    if (!emailVerified) return null
    if (!isAllowedEmail(email)) return null

    return email
  } catch (error) {
    return null
  }
}

function doPost (e) {
  const lock = LockService.getScriptLock()
  lock.tryLock(10000)

  try {
    const params = e.parameter || {}
    const verifiedEmail = verifyGoogleToken(params['google id token'])

    if (!verifiedEmail) {
      return jsonResponse({ result: 'error', error: 'unauthorized' })
    }

    const doc = SpreadsheetApp.openById(scriptProp.getProperty('key'))
    const sheet = doc.getSheetByName(sheetName)

    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]
    const nextRow = sheet.getLastRow() + 1

    const newRow = headers.map(function (rawHeader) {
      const header = String(rawHeader).trim()
      const normalizedHeader = header.toLowerCase()

      if (header === 'Date') return new Date()
      if (normalizedHeader === 'fecha') return params.fecha || new Date()
      if (normalizedHeader === 'usuario') {
        return params.usuario || params['usuario nombre'] || params['usuario email'] || ''
      }
      if (normalizedHeader === 'usuario email') return verifiedEmail

      return params[header] || params[rawHeader] || ''
    })

    sheet.getRange(nextRow, 1, 1, newRow.length).setValues([newRow])

    return jsonResponse({ result: 'success', row: nextRow })
  } catch (error) {
    return jsonResponse({ result: 'error', error: error.toString() })
  } finally {
    lock.releaseLock()
  }
}
